/**
 * 简单字符串哈希函数
 */
function hashString(str: string): string {
  let hash: any = 0;
  for (let i: any = 0; i < str.length; i++) {
    const char: any = str.charCodeAt(i);
    hash = ((hash << 5) - hash) + char;
    hash = hash & hash; // 转换为32位整数
  }
  return Math.abs(hash).toString(36);
}
